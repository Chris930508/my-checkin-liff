async function main() { try { // 1. 初始化 LIFF await liff.init({ liffId: "你的_LIFF_ID" }); if (!liff.isLoggedIn()) { liff.login(); } else { // 2. 取得定位 navigator.geolocation.getCurrentPosition(async function(pos) { const lat = pos.coords.latitude; const lng = pos.coords.longitude; try { // 3. 🚀 關鍵：使用 await 確保訊息「傳完」才關窗 await liff.sendMessages([{ type: 'text', text: `打卡座標:${lat},${lng}` }]); // 傳送成功後才關閉 liff.closeWindow(); } catch (sendError) { // 如果沒開 Scopes，這裡會跳錯誤 alert("傳送失敗，請檢查 Scopes: " + sendError.message); } }, function(err) { alert("GPS 錯誤: " + err.message); }, { enableHighAccuracy: true }); } } catch (initError) { alert("初始化失敗: " + initError.message); } } main();
