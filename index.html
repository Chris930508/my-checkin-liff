async function main() { const statusEl = document.getElementById('status'); try { statusEl.innerText = "正在初始化 LIFF..."; // 🚀 請確認這裡的 ID 是否正確 await liff.init({ liffId: "你的_LIFF_ID" }); statusEl.innerText = "LIFF 初始化成功，正在抓取 GPS..."; if (!liff.isLoggedIn()) { liff.login(); } else { navigator.geolocation.getCurrentPosition(async function(pos) { const lat = pos.coords.latitude; const lng = pos.coords.longitude; statusEl.innerText = `抓到座標：${lat}, ${lng}`; await liff.sendMessages([{ type: 'text', text: `打卡座標:${lat},${lng}` }]); liff.closeWindow(); }, function(err) { statusEl.innerText = "❌ GPS 失敗：" + err.message; }); } } catch (e) { // 🚀 如果初始化失敗，這裡會顯示原因 statusEl.innerHTML = `❌ 初始化失敗
${e.code}: ${e.message}`; console.error(e); } } main();
